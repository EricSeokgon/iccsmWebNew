나. 적용예제

 

import java.math.BigInteger;

import com.gpki.gpkiapi.GpkiApi;

import com.gpki.gpkiapi.cert.X509Certificate;

import com.gpki.gpkiapi.crypto.PrivateKey;

import com.gpki.gpkiapi.tsa.TimeStamp;

import com.gpki.gpkiapi.tsa.TimeStampToken;

import com.gpki.gpkiapi.util.Dump;

import com.gpki.gpkiapi.storage.Disk;

 

public class TimeStampService {

            

/*******************************************************************************************/

// 아래 define은 적용 시스템에 맞게 수정해 주어야 한다.

             private static String WORK_DIR_PATH                                  = ".";

             private static String DOCUMENT_FILE_PATH                         = "./document.txt";

             private static String CLIENT_SIGN_CERT_PATH                   

= "C:/GPKI/Certificate/class2/085사용자003_sig.cer";

             private static String CLIENT_SIGN_PRIKEY_PATH   

= "C:/GPKI/Certificate/class2/085사용자003_sig.key";

             private static String CLIENT_SIGN_PRIKEY_PASSWD           = "sppo1234";

             private static String TSA_SERVER_IP                                    = "152.99.56.61";

             private static int   TSA_SERVER_PORT                    = 80;

/*******************************************************************************************/

 

             void obtainTimeStampToken() {

 

                           try {

                                        // API 초기화

                                        GpkiApi.init(WORK_DIR_PATH);

            

                                        // 시점확인 토큰을 발급 받을 전자문서를 획득한다.

                                        byte[] bDocument = Disk.read(DOCUMENT_FILE_PATH);

                                       

                                        // 전자문서에 대한 시점확인 토큰을 요청한다.

                                        TimeStamp timeStamp = new TimeStamp();

                                        timeStamp.setMessage(bDocument);

                                        TimeStampToken timeStampToken =

timeStamp.reqTimeStampToken(TSA_SERVER_IP, TSA_SERVER_PORT);

                                       

                                        // 인증서 검증 시 사용할  서명용 인증서와 개인키를 획득

                                        X509Certificate signCert = Disk.readCert(CLIENT_SIGN_CERT_PATH);

                                        PrivateKey signPriKey = Disk.readPriKey(CLIENT_SIGN_PRIKEY_PATH,

CLIENT_SIGN_PRIKEY_PASSWD);

                                       

                                        // 시점확인 서버 인증서 검증 (3.2.5. 참조)

                                        X509Certificate tsaCert = timeStampToken.getTSACert();

                                       

                                        Cert cert = new Cert();

                                        cert.setMyCert(signCert, signPriKey);

                                        cert.setCert("tsa", tsaCert);

                                       

                                        cert.verify();

                                       

                                        // 받은 시점확인 토큰의 정보를 확인한다.

                                       

                                        System.out.println("* 시점확인 토큰 정보 ");

                                       

                                        // 1. 시점확인 서버의 이름

                                        System.out.println("  [TSA 서버 DN] " + tsaCert.getSubjectDN());

                                       

                                        // 2. 시점확인 토큰의 일련번호

                                        BigInteger serialNum = timeStampToken.getSerialNumber();

                                       

                                        System.out.print("  [토큰 일련번호] " + serialNum + "(");

                                       

                                        byte[] bBuf = serialNum.toByteArray();

                                        for (int i=0; i<bBuf.length; i++)

                                                     System.out.print(Dump.getHexString(bBuf[i]));

                                       

                                        System.out.println(")");

                                       

                                        // 시점확인 토큰 생성시간

                                        System.out.println("  [토큰 생성시간] " + timeStampToken.getGeneratedTime());

                                       

                                        // 시점확인 토큰 발급 정책

                                        System.out.println("  [토큰 발급정책] " + timeStampToken.getPolicy());

                                       

                                        // 시점확인 토큰을 요청한 메시지의 해쉬값  생성을 위해서 사용된 알고리즘

                                        System.out.println("  [해쉬 알고리즘] " + timeStampToken.getHashAlgorithm());

                                       

                                        // 시점확인 토큰을 요청한 메시지의 해쉬값

                                        System.out.print("  [메시지 해쉬값] ");

                                       

                                        bBuf = timeStampToken.getMessageImprint();

                                        for (int i=0; i<bBuf.length; i++)

                                                     System.out.print(Dump.getHexString(bBuf[i]));

                                       

                                        System.out.println(" ");

                          

                                        // 시점확인 토큰을 요청을 위해서 사용된 랜덤

                                        System.out.print("  [Nonce] ");

                                       

                                        bBuf = timeStampToken.getNonce();

                                        for (int i=0; i<bBuf.length; i++)

                                                     System.out.print(Dump.getHexString(bBuf[i]));

                                       

                                        System.out.println(" ");

                                       

                           } catch (Exception e) {

                                        e.printStackTrace();                     

                           }

             }

}
